	file.link AS "기도 제목"
	, 기도-대상 AS "기도 대상"
	, 기도-내용 AS "기도 내용"
FROM "10. 🌏Atlas/102. 💡내 생각/1021. †신앙"
WHERE 응답 = false
AND 기도-기간-지남 != true
SORT 기도-대상, 기도-내용, file.link
